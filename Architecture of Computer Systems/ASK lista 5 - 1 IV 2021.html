Zadanie 1

Przeanalizujmy po kolei tę procedurę.
W linijce 1 sprawdzamy czy wartość w rejestrze %esi jest równa 0 i jeśli tak, to zwracamy 0.
Xorujemy %edx ze sobą i eax ze sobą. Czyli zerujemy wartości w tych rejestrach. Przenosimy wartość z %edi do %ecx i andujemy to z 1ką (żeby dostać ostatni bit?). Dodajemy ten bit do %eax. Shiftujemy %rdi (czyli naszego x o jeden bit w prawo arytmetycznie).

Patrzymy czy bit jest zapalony, jeśli tak, to zwiększamy edx. Procedura oblicza ile zapalonych jest pierwszych n bitów od prawej do lewej.

W języku C:













int puzzle( long x, unsigned n ){
    if( n == 0 ) 
        return 0;
    
    int sum = 0;
    
    for( int i = 0; i < n; i++ ){
        sum += x & 1;
        x = x >> 1;
    }
    return sum;
}

Zadanie 2


s -> %rdi
d -> %rsi

W %raxie ląduje %rdi, czyli s.
Do %r9b wkładamy wartość z raxa, czyli pierwszy char z s, to %r8 ląduje następny adres.
Przesuwamy rsi do rdx (czyli w rdxie ląduje string d). Przesuwamy pierwszego chara z d to cl. Zwiększamy rdx. Patrzymy czy cl == 0 (w przypadku chara cl == ‘\0’ -> koniec stringa).
Jeśli tak, to wyskakujemy do L4 czyli z raxa odejmujemy rdi (czyli tak naprawdę ile iteracji dużej pętli zostało zrobione w raxie jest adres na obecny adres).
Inaczej sprawdzamy czy one są różne, jeśli tak, to wskakujemy do L2 czyli do jakiejś wewnętrzenej pętli.
Jeśli nie, to przenosimy wartość z raxa do r8 (czyli następny element z s!). Przeskakujemy bezwarunkowo do L3.















puzzle2:  movq %rdi, %rax          ;B1
.L3:      movb (%rax), %r9b        ;B2
          leaq 1(%rax), %r8
          movq %rsi, %rdx
.L2:      movb (%rdx), %cl         ;B3
          incq %rdx
          testb %cl, %cl
          je .L4                   
          cmpb %cl, %r9b           ;B4
          jne .L2                   
          movq %r8, %rax           ;B5
          jmp .L3                  
.L4:      subq %rdi, %rax          ;B6
          ret       




























// char *t1 -> %r9b, char *t2 -> %r8, char *t3 -> %rdx, char t4 -> %cl
long puzzle2(char *s, char *d)
{
    char *result = s;
    while (1)
    {
        char t1 = *pCurrS;
        char *t2 = pCurrS+1;
        char *t3 = d;
        char t4;
        do
        {
            t4 = *t3;
            t3++;
            if (t4 == '\0')
                return (long)(pCurrS - s);
                
        } while (t1 != t4);
        pCurrS = t2;
    }
}

Zadanie 3

Zerujemy pierwsze 32 bity liczby w %edi (czyli n).
Shiftujemy d o 32 bity w lewo. d <<= 32.
W %edx ląduje liczba 32.
Do %ecx 0x80000000 czyli 1 << 31.
Xorujemy %eax ze sobą, czyli zerujemy go.
%rdi mnożymy razy dwa. Do r8 leci wynik mnożenia, odejmujemy od rsi r8 (czyli d - %rdi).
Jeśli różnica jest ujemna, to lądujemy w linii 13, jeśli nie, to ORujemy ecx z eaxem i przenosimy %rdi do %r8.

Shiftujemy %ecx o jeden w lewo, edx zmniejszamy o 1 i patrzymy czy powinniśmy wyjść z pętli, jeśli tak, to wychodzimy z programu, jak nie, to wracamy do mnożenia rdi przez 2.
















puzzle3: movl %edi, %edi                ;B1
         salq $32, %rsi
         movl $32, %edx
         movl $0x80000000, %ecx
         xorl %eax, %eax
.L3:     addq %rdi, %rdi                ;B2
         movq %rdi, %r8
         subq %rsi, %r8
         js .L2                         ;B3
         orl %ecx, %eax
         movq %r8, %rdi
.L2:     shrl %ecx                      ;B4
         decl %edx
         jne .L3                        
         ret                            ;B5























uint32_t puzzle3 (uint32_t n, uint32_t d){
    uint64_t ext_n = n;
    uint64_t ext_d = (uint64_t)d <<32;
    uint32_t res = 0;
	uint32_t bit = 0x80000000;
    for (int i = 32; i > 0; i--) {
        ext_n += ext_n;
        uint64_t diff = ext_n - ext_d;
        if ((int64_t)diff >= 0){
            res |= bit;
            ext_n = diff;
        }
        bit >>= 1;
    }
    return res;
}

Zadanie 4


a -> %rdi
v -> %rsi
s -> %rdx
e -> %rcx

Przenosimy wartość s do do raxa, odejmujemy od niej e, dzielimy wszystko przez 2 i dodajemy wartość rax do rdx.
Czyli w rax znajduje sie (e-s)/2 + s. Porównujemy rdx < rcx, jeśli tak to wtychdozimy z programu ze zwrotem -1.
Jeśli nie to do %r8 przenosimy wartość z %rdi + %rax * 8, czyli tak naprawdę wartość ze środka tablicy danej wskaźnikiem a.
Jeśli v == tej wartości, to wychodzimy z indeksem w raxie.

Wpp jeśli v < tej wartości, to do rdx (start) przenosimy %rax + 1, czyli środek podtablicy + 1 i wywołujemy procedurę z takimi argumentami:
a -> a
v -> v
s -> (e-s)/2 + s + 1
e -> e
Wpp za e wstawiamy środek - 1, czyli (e-s)/2 + s - 1 i wywołujemy.

Bloki podstawowe:





















puzzle4:
movq %rcx, %rax                            ;B1
subq %rdx, %rax
shrq %rax
addq %rdx, %rax
cmpq %rdx, %rcx
jb .L5                                      
movq (%rdi,%rax,8), %r8                    ;B2
cmpq %rsi, %r8
je .L10                                    
cmpq %rsi, %r8                             ;B3
jg .L11                                    
leaq 1(%rax), %rdx                         ;B4
call puzzle4        
.L10: ret                                  ;B5
.L11: leaq -1(%rax), %rcx                  ;B6
call puzzle4       
ret          
.L5: movl $-1, %eax                        ;B7
ret

Kod w C:

















int puzzle4 (long* a, long v, uint64_t s, uint64_t e)
{
    if (e < s){
        return -1;
    } 
    uint64_t mid = (e-s)/2 + s;
    long midEl = a[mid];
    
    if ( midEl == v ){
        return mid;
    } 
    if ( v < mid ){
        return puzzle4( a, v, s, mid - 1 );
    } 
    return puzzle4( a, v, mid + 1, e );
}

Zadanie 5


\(0x3c = 60\)
W pierwszej linijce odejmujemy od n liczbę 60.
Później sprawdzamy czy n - 60, porównujemy to do 5.



switch (n - 60){
}

Jeśli , to skaczemy pod adres defaulta (czyli 0x4005c3). W przeciwnym razie obliczamy adres, pod który mamy skoczyć (czyli patrzymy pod indeks tablicy skoków).
Jeśli n - 60 == 0, to pod 0x4005a1.
Jeśli n - 60 == 1, to pod 0x4005a1. Dwa razy skaczemy więc pod ten sam adres, czyli case 0 jest tak naprawdę pusty.
Do rejestru %rax wkładamy 8 * rdi czyli 8 * x.

Jeśli n - 60 == 2, to skaczemy pod 0x4005b2, czyli w linię 11. Przenosimy wartość x do raxa, shiftujemy w lewo o 16 i odejmujemy x. Czyli jest to tak naprawdę mnożenie przez 15.
Później to część kolejnego case’a.



case 2:
    x *= 15;

Jeśli n - 60 == 3, to skaczemy pod defaulta. Czyli tego case’a można ominąć.

Jeśli n - 60 == 4, to lądujemy w linii 8. Przenosimy wartość z rdi do raxa, shiftujemy ją w prawo o 3 i od razu zwracamy.



case 4:
    return x << 3;

Jeśli n - 60 == 5, to mnożymy x przez samego siebie.



case 5:
    x *= x;

Na końcu do raxa przenosimy wartość z %rdi + 0x4b, czyli %rdi + 75.

Zatem, składając wszystko do jednej procedury, w języku C wygląda to tak:
















long switch_prob(long x, long n){
    switch (n - 60){
        case 0:
        case 1:
            return 8 * x;
        case 4:
            return x >> 3;
        case 2:
            x *= 15;
        case 5:
            x *= 5;
        default:
            return x + 75;    
    }
}
